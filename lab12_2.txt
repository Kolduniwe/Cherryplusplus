#include <iostream>

using namespace std;

int main() {
    // Оголошення двовимірного масиву 10x10
    int matrix[10][10] = {0};
    
    // Вказівник на перший елемент масиву
    int *ptr = &matrix[0][0];
    
    // Заповнюємо масив таблицею множення
    for (int i = 0; i < 10; i++) {
        for (int j = 0; j < 10; j++) {
            *(ptr + i * 10 + j) = (i + 1) * (j + 1);  // Записуємо значення таблиці множення
        }
    }

    // Виведення масиву
    for (int i = 0; i < 10; i++) {
        for (int j = 0; j < 10; j++) {
            cout << *(ptr + i * 10 + j) << "\t";  // Виводимо елемент через вказівник
        }
        cout << endl;
    }

    return 0;
}
